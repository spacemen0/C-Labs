#include <iostream>
#include <string>
#include <cstdlib>
#include <ctime>

std::string generateRandomWord() {
    std::string word;
    int length = rand() % 10 + 1; // Generate a random word length between 1 and 10

    for (int i = 0; i < length; i++) {
        char c = 'a' + rand() % 26; // Generate a random lowercase letter
        word += c;
    }

    return word;
}

int main() {
    srand(time(0)); // Seed the random number generator with current time

    for (int i = 0; i < 10; i++) {
        std::cout << generateRandomWord() << std::endl;
    }

    return 0;
}
